JavaScriptとは？ / JavaScriptの基礎

JavaScriptは、ウェブサイトに動きを与える言語です。たとえば、ボタンをクリックしたときの反応や、ランダムな表示、リスト管理など、Web体験を豊かにします。

変数と定数について

変数（let）は値を後から変えることができ、定数（const）は一度決めたら変更できません。

// 名前を格納する変数を定義
let name = "太郎";
// 年齢を格納する定数を定義
const age = 20;
// 出力エリアに名前と年齢を表示
document.getElementById("output1").textContent = `名前: ${name}、年齢: ${age}`;

関数とは？

関数は一連の処理をまとめたものです。必要なときに呼び出して使います。

// あいさつする関数を定義
function sayHello() {
  alert("こんにちは！"); // アラート表示
}
// 関数を呼び出し
sayHello();

条件分岐とは？

条件によって処理を変えることができます。if文を使います。

// スコアを定義
let score = 85;
let message = "";
// 条件分岐：スコアが80以上かどうかで結果を変える
if (score >= 80) {
  message = "よくできました";
} else {
  message = "がんばりましょう";
}
// 結果を出力
document.getElementById("output3").textContent = message;

イベントとDOM

イベントとは、ユーザーの操作（例：クリック）です。DOMはHTMLの構造を表し、JavaScriptで操作可能です。

// demoBtnがクリックされたらアラート表示
document.getElementById("demoBtn").addEventListener("click", function() {
  alert("ボタンがクリックされました");
});

おみくじ機能

ランダムに「大吉」などの結果を表示する簡単な機能です。

// 結果リストを用意
const results = ["大吉", "中吉", "小吉", "凶"];
// ランダムなインデックスを取得
const index = Math.floor(Math.random() * results.length);
// 結果を表示
document.getElementById("omikujiResult").textContent = `結果: ${results[index]}`;

スライダー機能（画像の切り替え）

一定間隔で画像が自動で切り替わる仕組みです。

// 現在のスライド位置
let currentSlide = 0;

// スライダーを更新する関数
function updateSlider() {
  const slider = document.getElementById("slider");
  const slides = slider.querySelectorAll("img");
  const slideWidth = slider.clientWidth / slides.length;
  slider.style.transform = `translateX(-${currentSlide * slideWidth}px)`;
}

// 次のスライドに進む関数
function nextSlide() {
  const slider = document.getElementById("slider");
  const slides = slider.querySelectorAll("img");
  currentSlide = (currentSlide + 1) % slides.length;
  updateSlider();
}

// 一定時間ごとにスライドを切り替え
setInterval(nextSlide, 4000);
// 初期化とリサイズ時の調整
window.addEventListener("resize", updateSlider);
window.addEventListener("load", updateSlider);

ToDoリスト

やることを追加・削除できるリスト機能です。

    // ToDoを追加する関数
function addTodo() {
  const input = document.getElementById("todoInput");
  const value = input.value.trim();
  if (!value) return; // 入力が空なら終了

  const li = document.createElement("li");
  li.className = "todo-item";
  li.textContent = value;

  const delBtn = document.createElement("button");
  delBtn.textContent = "削除";
  delBtn.onclick = () => li.remove(); // ボタンを押したら削除

  li.appendChild(delBtn);
  document.getElementById("todoList").appendChild(li);
  input.value = ""; // 入力欄をリセット
}

    お絵描きボード

    マウスで自由に線を描いたり、消したりできるキャンバスです。


    // キャンバスと描画コンテキストを取得
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");
let drawing = false; // 描画中かどうかのフラグ

// マウスを押したとき、描画を開始
canvas.addEventListener("mousedown", e => {
  drawing = true;
  ctx.beginPath();
  ctx.moveTo(e.offsetX, e.offsetY);
});

// マウスを動かしたとき、線を描く
canvas.addEventListener("mousemove", e => {
  if (!drawing) return;
  ctx.lineTo(e.offsetX, e.offsetY);
  ctx.stroke();
});

// マウスを離したとき、描画終了
canvas.addEventListener("mouseup", () => (drawing = false));

// キャンバスを全消去する関数
function clearCanvas() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
}
