JavaScript 基本文法まとめ

console.log コメント 変数宣言 データ型 演算子 条件分岐 ループ 関数 スコープ ← ポートフォリオへ戻る

console.log（出力）

console.log('Hello World');

コメント

// これは1行コメント

  これは複数行コメント
*/

変数宣言（const, let）

const price = 500;
let count = 3;
// var は再定義・再代入ができるが非推奨

データ型（プリミティブ型）

  • String, Number, Boolean, undefined, null, Symbol
console.log(typeof 'Hi');     // "string"
console.log(typeof 123);     // "number"
console.log(typeof true);    // "boolean"
console.log(typeof undefined); // "undefined"
console.log(typeof null);     // "object" ← 歴史的仕様

演算子・注意点

console.log('5' * 3);  // 15
console.log('5' + 3);        // "53"

console.log(parseInt('5') + 3); // 8

条件分岐（if / switch／三項演算子）

if (score >= 80) {
  console.log('Great!');
} else if (score >= 60) {
  console.log('Good');
} else {
  console.log('OK...');
}

score >= 80 ? console.log('Great!') : console.log('OK...');

switch (signal) {
  case 'red': console.log('Stop!'); break;
  case 'blue':
  case 'green': console.log('Go!'); break;
  default: console.log('Wrong Signal');
}

ループ（for / while / do…while）

for (let i = 1; i <= 5; i++) {
  console.log(`${i}: Hello`);
}

let hp = 100;
while (hp > 0) {
  console.log(`HP: ${hp}`);
  hp -= 15;
}

let num = -10;
do {
  console.log(num);
  num += 5;
} while (num < 0);

関数宣言・関数式・アロー関数

// 関数宣言
function sum(a, b) {
  return a + b;
}

// 関数式
const mul = function(a, b) {
  return a * b;
};

// アロー関数
const double = a => a * 2;
console.log(sum(1,2), mul(2,3), double(5));

スコープ（有効範囲）

{
  const x = 10;
  console.log(x); // OK
}
console.log(x); // エラー：x is not defined

const global = 5;
function f() {
  const global = 3;
  console.log(global);
}
f();
console.log(global); // 5
